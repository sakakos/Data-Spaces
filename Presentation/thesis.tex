% ─── Διαφάνεια 3: Το πρόβλημα ────────────────────────────────────
\begin{frame}{Το Πρόβλημα του Διαμοιρασμού Δεδομένων}
Στη σύγχρονη εποχή, τα δεδομένα έχουν αναδειχθεί σε έναν από τους σημαντικότερους
παράγοντες προόδου και καινοτομίας. Ωστόσο, ο διαμοιρασμός δεδομένων μας εγείρει ερωτήματα:
\bigskip
\begin{itemize}
    \item Πως μπορεί ένας οργανισμός να μοιραστεί δεδομένα με τρίτους διατηρώντας τον έλεγχο
    επί της χρήσης τους.
    \item Πώς εξασφαλίζεται ότι ο αποδέκτης των δεδομένων είναι αξιόπιστος και θα χρησιμοποιήσει
    τα δεδομένα σύμφωνα με την βούληση του παρόχου.
    \item Πώς επιτυγχάνεται η διαλειτουργικότητα χωρίς κεντρική
    αποθήκευση.
\end{itemize}
\bigskip
Παραδοσιακά ένας ενδιάμεσος φορέας συγκέντρωνε και διαχειρίζοταν τα δεδομένα, αποτελώντας
σημείο αποτυχίας και θέτωντας ζητήματα ασφάλειας και εμπιστοσύνης.
\end{frame}

% ─── Διαφάνεια 4: Data Spaces ────────────────────────
\begin{frame}{Η Έννοια των \en{Data Spaces}}
Ο όρος \textlatin{Data Space} 
αναφέρεται σε ένα κατανεμημένο σύστημα ανταλλαγής δεδομένων που βασίζεται σε κοινά συμφωνημένα πρότυπα, πολιτικές και υποδομές εμπιστοσύνης,
επιτρέποντας σε οργανισμούς να μοιράζονται δεδομένα διατηρώντας παράλληλα την κυριαρχία
τους επί αυτών.
\bigskip

Βασικές αρχές:
\begin{itemize}
    \item \textbf{Ψηφιακή Κυριαρχία:} Κάθε οργανισμός διατηρεί τον πλήρη
    έλεγχο επί των δεδομένων του.
    \item \textbf{Αποκεντρωμένη Αρχιτεκτονική:} Τα δεδομένα παραμένουν
    στις υποδομές του παρόχου.
    \item \textbf{Διαλειτουργικότητα:} Κοινά πρότυπα και πρωτόκολλα
    επιτρέπουν την επικοινωνία μεταξύ ετερογενών συστημάτων.
    \item \textbf{Πλαίσιο Εμπιστοσύνης:} Τεχνικοί μηχανισμοί αντικαθιστούν την εμπιστοσύνη σε
    μεσάζοντα.
\end{itemize}
\end{frame}

\begin{frame}{Οι ρόλοι ενός \en{Data Space}}
Σύμφωνα με τη \en{Data Spaces Business Alliance (DSBA)}, οι συμμετέχοντες 
διακρίνωνται στους ακόλουθους ρόλους:
\bigskip
\begin{itemize}
    \item \textbf{Πάροχος Δεδομένων:} Ο οργανισμός που κατέχει και διαθέτει τα δεδομένα
    (\textlatin{EU.EORI.NLPLEGMA}).
    \item \textbf{Καταναλωτής Δεδομένων:} Ο οργανισμός που απαιτεί πρόσβαση στα δεδομένα
    του παρόχου (\textlatin{EU.EORI.NLPLEGMACONSUMER}).
    \item \textbf{Αρχή Εμπιστοσύνης:} Ο φορέας που διαχειρίζεται το μητρώο των εγκεκριμένων
    συμμετχόντων και επαληθεύει την εγκυρότητα των ψηφιακών τους πιστοποιητικών
    (\textlatin{iSHARE Satellite}).
    \item \textbf{Μητρώο Εξουσιοδότησης:} Ο φορέας που αποθηκεύει και διαχειρίζεται τις
    πολιτικές εξουσιόδοτησης (\textlatin{Keyrock Authorization Registry}).
\end{itemize}
\end{frame}

\begin{frame}{Αντικείμενο της Εργασίας}
Η παρούσα διπλωματική αποτελεί έναν πειραματισμό με το υπάρχον λογισμικό για \textlatin{Data Spaces}
με σκοπό να το να το εμπλουτίσει με ανοικτές τεχνολογίες για να επιτευχθεί η ψηφιακή κυριαρχία.
\bigskip

\begin{itemize}
    \item Χρήση των ανοιχτών τεχνολογιών της \en{FIWARE Foundation}.
    \item Εφαρμογή του πλαισίου εμπιστοσύνης \en{i4Trust / iSHARE}.
    \item Μοντελοποίηση πραγματικών δεδομένων σε \en{NGSI-LD} με χρήση
    \en{Smart Data Models}.
\end{itemize}
Στόχος είναι να παρουσιαστεί ένας πλήρης κύκλος εκχώρησης πρόσβασης σε δεδομένα ενός παρόχου
από τον καταναλωτή δεδομένων.
\end{frame}

\begin{frame}{Ψηφιακά Πιστοποιητικά \en{X.509}}
Ένα πιστοποιητικό \textlatin{X.509} συνδέει το δημόσιο κλειδί με μία ταυτότητα,
εγγύηση που δίνει μία Αρχή Πιστοποίησης (\en{Certificate Authority}).

\begin{table}
    \centering
    \small
    \begin{tabular}{@{} l p{7.5cm} @{}}
        \toprule
        Εκδότης & Ποια \textlatin{CA} υπογράφει.\\
        \addlinespace
        Περίοδος Ισχύος & Το χρονικό παράθυρο εγκυρότητας. \\
        \addlinespace
        \textlatin{keyUsage} & Καθορίζει επιτρεπόμενες χρήσεις του κλειδιού. \\
        \addlinespace
        Υπογραφή & Η ψηφιακή υπογραφή πάνω σε όλα τα πεδία. \\
        \bottomrule
    \end{tabular}
\end{table}
\bigskip

\textbf{Αλυσίδα Εμπιστοσύνης(\textlatin{Chain of Trust})}:
Μία \textlatin{CA} μπορεί να είναι πιστοποιημένη από από μία ανώτερη \textlatin{CA}.
  
\textbf{Ψηφιακό Αποτύπωμα (\textlatin{Fingerprint})}:
Το αποτέλεσμα της συνάρτησης \textlatin{SHA-256} επί του πιστοποιητικού.
\end{frame}


\begin{frame}[fragile]{\en{JSON Web Tokens (JWT)}}
Τα \en{JWT} είναι αυτόνομα \en{tokens} που
μεταφέρουν πληροφορίες υπογεγραμμένες ψηφιακά. Αποτελούνται από
τρία τμήματα: επικεφαλίδα, ωφέλιμο φορτίο, υπογραφή.
\bigskip

Το πλαίσιο \en{iSHARE} επεκτείνει τα \en{JWT} με δύο σημαντικές
προσθήκες:

\selectlanguage{english}
\begin{lstlisting}
Header:  { "alg": "RS256", "typ": "JWT",
           "x5c": ["<cert_leaf>", "<cert_CA>"] }

Payload: { "iss": "EU.EORI.NLPLEGMA",
           "sub": "EU.EORI.NLPLEGMA",
           "aud": "EU.EORI.NLPLEGMA",
           "iat": 1716000000,
           "exp": 1716000030 }
\end{lstlisting}
\selectlanguage{greek}

\textlatin{\textlatin{RSASHA256(base64url(header) + "." + base64url(payload), private\_key)}} \\
Συνήθως ένα \en{JWT} έχει διάρκεια ζωής 30 δευτερόλεπτα.
\end{frame}

% ─── NGSI-LD ────────────────────────────────────────
\begin{frame}{Το Πρότυπο \en{NGSI-LD}}
Το \en{NGSI-LD} (\en{Next Generation Service Interfaces - Linked Data}) είναι το πρότυπο \en{API}
για την ανταλλαγή πληροφοριών μεταξύ συστημάτων. \\

To \textlatin{JSON-LD} είναι ο τρόπος που το \en{NGSI-LD} αναπαριστά τα δεδομένα.
Kάθε οντότητα και κάθε ιδιότητά της έχει μία παγκόσμια μοναδική ταυτότητα (\en{URI}).
\bigskip

Μοντέλο πληροφορίας:
\begin{itemize}
    \item \textbf{Οντότητα (\en{Entity})}: Πραγματικό ή εννοιολογικό αντικείμενο.
    \item \textbf{Ιδιότητα (\en{Property})}: Μετρήσιμο ή περιγραφικό χαρακτηριστικό.
    \item \textbf{Σχέση (\en{Relationship})}: Σύνδεση μεταξύ οντοτήτων.
\end{itemize}
\bigskip

Το πεδίο \textlatin{@context} αντιστοιχίζει τα ονόματα
ιδιοτήτων σε παγκόσμια \en{URI}, δίνοντας στα δεδομένα σημασιολογική ερμηνεία.
\end{frame}

    \item Παρέχει τη λίστα εμπιστευμένων \en{CA} (\textlatin{trusted list}).
    \item Επαληθεύει την κατάσταση συμμετεχόντων μέσω του

\bigskip

\textbf{\en{Keyrock Identity Manager} \en{Authorization Registry}:}
\begin{itemize}
    \item Είναι ταυτόχρονα πάροχος ταυτότητας \textit{και} μητρώο

    \item Δέχεται \en{iSHARE JWT} και εκδίδει \en{access tokens}.
    \item Διαχειρίζεται τις πολιτικές εκχώρησης μέσω της βάσης
    \en{MySQL}.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Το \en{Plegma Dataset}}
Πραγματικά δεδομένα από 13 νοικοκυριά στην Ελλάδα, διαθέσιμα από

    \item Μεταδεδομένα κτιρίου και ανωνυμοποιημένα δημογραφικά

\end{itemize}

\begin{frame}{Σενάρια Επίδειξης Ελέγχου Πρόσβασης}
Η πειραματική αξιολόγηση έγινε μέσω τριών διακριτών σεναρίων:
\bigskip
\begin{enumerate}
    \item \textbf{Πρόσβαση χωρίς ταυτοποίηση}: επαληθεύει ότι το
    σύστημα απορρίπτει μη εξουσιοδοτημένα αιτήματα.
    \item \textbf{Πρόσβαση με \en{token} παρόχου}: επαληθεύει
    την πλήρη πρόσβαση του παρόχου στα δεδομένα του.
    \item \textbf{Πρόσβαση με \en{token} καταναλωτή}: επαληθεύει
    τον πλήρη κύκλο εκχώρησης εξουσιοδότησης.
\end{enumerate}
\bigskip
Τα σενάρια εκτελούνται μέσω \en{scripts} που αυτοματοποιούν τη δημιουργία \en{tokens} και την αποστολή αιτημάτων.
\end{frame}

\begin{frame}[fragile]{Κύκλος Εκχώρησης Πρόσβασης Καταναλωτή}

Δημιουργία \en{iSHARE JWT} καταναλωτή:
\selectlanguage{english}
\begin{lstlisting}
$CONSUMER_JWT = docker run --rm -v "${ISHARE_DIR}:/iShare" python:3.9-slim
    sh -c "pip install PyJWT cryptography -q &&
           python3 /iShare/gen_consumer_token.py"
\end{lstlisting}
\selectlanguage{greek}

Ανταλλαγή \en{JWT} με \en{access token} από τον \en{Keyrock}:
\selectlanguage{english}
\begin{lstlisting}
curl -X POST http://localhost:3007/oauth2/token
    -d "grant_type=client_credentials
        &scope=iSHARE
        &client_assertion_type=...jwt-bearer
        &client_assertion=$CONSUMER_JWT
        &client_id=EU.EORI.NLPLEGMACONSUMER"
\end{lstlisting}
\selectlanguage{greek}

Πρόσβαση στα δεδομένα μέσω \en{Kong}:
\selectlanguage{english}
\begin{lstlisting}
curl -H "Authorization: Bearer $ACCESS_TOKEN"
    "http://localhost:8000/ngsi-ld/v1/entities
     ?type=...HouseholdElectricMeasurement&limit=1"
\end{lstlisting}
\selectlanguage{greek}
Το \en{Kong} αξιολογεί το \en{delegationEvidence} και επιτρέπει την πρόσβαση.
\end{frame}

\begin{frame}{Συνεισφορά της Εργασίας}
Συνολικά, αυτός ο πειραματισμός με το υπάρχον λογισμικό για \en{Data Spaces} επιτυγχάνει:
\bigskip
\begin{itemize}
    \item Χρήση ανοικτών τεχνολογιών για την αντιγραφή της αρχιτεκτονικής ενός
    \en{Data Space}, χωρίς να χρειαστεί κανείς να αξιοποιήσει τον \en{FIWARE Data Space Connector}.
    \bigskip

    \item Αυτοματοποιημένα \en{scripts} στησίματος που επιτρέπουν
    την αναπαραγωγή της υλοποίησης.
    \bigskip

    \item Τεκμηρίωση πρακτικών προκλήσεων (μορφές κλειδιών, σειρά
    αρχικοποίησης, αντιστοίχιση \en{UUID}/\en{EORI}).
    \bigskip

\end{itemize}
\end{frame}

\begin{frame}{Περιορισμοί και Μελλοντικές Επεκτάσεις}
Η παρούσα υλοποίηση έχει περιορισμούς που πρέπει να ληφθούν υπόψη:
\smallskip
\begin{itemize}
    \item \textbf{Τοπικό περιβάλλον:} Εκτελείται σε ένα μηχάνημα
    μέσω \en{Docker Compose} και δεν υλοποιεί ανεξάρτητα \en{API}.
    \bigskip

    \item \textbf{Ανάπτυξη σε \en{Kubernetes}:} Μεταφορά σε
    περιβάλλον \en{cloud} με χρήση \en{Helm charts} της
    \en{FIWARE Foundation}.
    \bigskip

    \item \textbf{Ανακάλυψη Δεδομένων:} Προσθήκη μηχανισμού
    αυτόματης αναζήτησης διαθέσιμων \en{datasets} από
    καταναλωτές.
    \bigskip

    \item \textbf{\textlatin{Hardcoded consumer:}} Θα πρέπει να δημιουργηθεί \en{API}
    το οποίο θα καλεί όποιος οργανισμός επιθυμεί να γίνει καταναλωτής.
    \bigskip
\end{itemize}
\end{frame}
